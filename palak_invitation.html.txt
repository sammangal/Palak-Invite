/* CALENDAR */
function openCalendar() {
  document.getElementById("calendarBox").style.display = "block";
}

/* CONFIRM */
function confirmBooking() {
  const date = document.getElementById("datePicker").value;

  if (!date) {
    alert("Please select a date first 🙂");
    return;
  }

  document.getElementById("confirmation").style.display = "block";

  const sig = document.getElementById("signature");
  sig.style.display = "block";
  setTimeout(() => sig.style.opacity = 1, 100);

  /* silent mail trigger */
  const mailLink = "mailto:sammangal1996@gmail.com?subject=Celebration Appointment&body=Chosen Date: " + date;
  const a = document.createElement('a');
  a.href = mailLink;
  a.click();
}

/* FUN BUTTON */
function funMsg() {
  alert("Pinky promise noted. Radiology level commitment recorded 💖");
}

/* DODGING BUTTON */
const btn = document.getElementById("declineBtn");

btn.addEventListener("mouseover", () => {
  const jokes = [
    "Diagnostic error detected",
    "Radiology report inconclusive",
    "Invalid selection",
    "Option unavailable",
    "Please reassess findings"
  ];

  btn.innerText = jokes[Math.floor(Math.random() * jokes.length)];

  const x = Math.random() * (window.innerWidth - 150);
  const y = Math.random() * (window.innerHeight - 80);

  btn.style.position = "absolute";
  btn.style.left = `${x}px`;
  btn.style.top = `${y}px`;
});
